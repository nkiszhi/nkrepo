# ============================================================================
# Installation Notes
# ============================================================================
#
# Ubuntu 22.04 Quick Setup:
# -------------------------
# 1. Install system dependencies:
#    sudo apt update
#    sudo apt install -y python3 python3-pip python3-venv python3-dev \
#      build-essential libssl-dev libffi-dev \
#      libmysqlclient-dev pkg-config \
#      libfuzzy-dev libyara-dev
#
# 2. Create virtual environment:
#    python3 -m venv venv
#    source venv/bin/activate
#
# 3. Install Python packages:
#    pip install --upgrade pip setuptools wheel
#    pip install -r requirements.txt
#
# Windows Quick Setup:
# --------------------
# 1. Install Python 3.8+ from https://www.python.org/
#
# 2. Install Visual C++ Build Tools (for ssdeep):
#    https://visualstudio.microsoft.com/downloads/
#
# 3. Create virtual environment:
#    python -m venv venv
#    venv\Scripts\activate.bat
#
# 4. Install Python packages:
#    pip install --upgrade pip setuptools wheel
#    pip install -r requirements.txt
#
# GPU Support (Optional):
# -----------------------
# For CUDA 12.1 (NVIDIA GPU):
#    pip install torch torchvision --index-url https://download.pytorch.org/whl/cu121
#
# For CUDA 11.8:
#    pip install torch torchvision --index-url https://download.pytorch.org/whl/cu118
#
# For CPU only (smaller download):
#    pip install torch torchvision --index-url https://download.pytorch.org/whl/cpu
#
# Troubleshooting:
# ----------------
# If ssdeep installation fails:
#   Ubuntu: sudo apt install libfuzzy-dev
#   Windows: Download prebuilt wheel from https://www.lfd.uci.edu/~gohlke/pythonlibs/
#
# If yara-python installation fails:
#   Ubuntu: sudo apt install libyara-dev
#   Windows: pip install yara-python-wheels
#
# If PyMySQL installation fails:
#   Ubuntu: sudo apt install libmysqlclient-dev python3-dev
#   Windows: Ensure Visual C++ Build Tools are installed
#
# Memory issues during pip install:
#   export PIP_NO_CACHE_DIR=1
#   pip install -r requirements.txt --no-cache-dir
#
